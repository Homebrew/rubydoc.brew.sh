def [](key)
  return unless tap.git?
  return unless Utils::Git.available?

  tap.path.cd do
    Utils.popen_read("git", "config", "--get", "homebrew.#{key}").chomp.presence
  end
end

def []=(key, value)
  return unless tap.git?
  return unless Utils::Git.available?

  tap.path.cd do
    safe_system "git", "config", "--replace-all", "homebrew.#{key}", value.to_s
  end
end
