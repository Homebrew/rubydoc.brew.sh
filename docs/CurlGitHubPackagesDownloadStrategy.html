def initialize(url, name, version, **meta)
  meta[:headers] ||= []
  # GitHub Packages authorization header.
  # HOMEBREW_GITHUB_PACKAGES_AUTH set in brew.sh
  meta[:headers] << "Authorization: #{HOMEBREW_GITHUB_PACKAGES_AUTH}"
  super(url, name, version, **meta)
end
