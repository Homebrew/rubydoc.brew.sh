def call
  if @uri
    result = curl_output("--fail", "--silent", "--location", @uri)
    result.assert_success!

    page = result.stdout
    page.extend PageWithURL
    page.url = URI(@uri)

    instance_exec(page, &@block)
  else
    instance_exec(&@block)
  end
end
