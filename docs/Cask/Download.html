def downloader
  @downloader ||= begin
    strategy = DownloadStrategyDetector.detect(cask.url.to_s, cask.url.using)
    strategy.new(cask.url.to_s, cask.token, cask.version, cache: Cache.path, **cask.url.specs)
  end
end
