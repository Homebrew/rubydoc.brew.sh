def self.reason(name)
  case name
  when /^adobe-(after|illustrator|indesign|photoshop|premiere)/
    "Adobe casks were removed because they are too difficult to maintain."
  when /^pharo$/
    "Pharo developers maintain their own tap."
  end
end
