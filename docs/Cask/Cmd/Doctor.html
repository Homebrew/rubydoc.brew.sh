def run
  require "diagnostic"

  success = true

  checks = Homebrew::Diagnostic::Checks.new(verbose: true)
  checks.cask_checks.each do |check|
    out = checks.send(check)

    if out.present?
      success = false
      puts out
    end
  end

  raise CaskError, "There are some problems with your setup." unless success
end
