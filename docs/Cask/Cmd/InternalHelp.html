def run
  max_command_len = Cmd.commands.map(&:length).max
  puts "Unstable Internal-use Commands:\n\n"
  Cmd.command_classes.each do |klass|
    next if klass.visible?

    puts "    #{klass.command_name.ljust(max_command_len)}  #{klass.help}"
  end
  puts "\n"
end
