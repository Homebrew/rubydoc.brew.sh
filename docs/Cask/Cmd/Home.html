def run
  if casks.none?
    odebug "Opening project homepage"
    self.class.open_url "https://brew.sh/"
  else
    casks.each do |cask|
      odebug "Opening homepage for Cask #{cask}"
      self.class.open_url cask.homepage
    end
  end
end
