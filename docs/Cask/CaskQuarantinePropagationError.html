def to_s
  s = +"Failed to quarantine one or more files within #{path}."

  unless reason.empty?
    s << " Here's the reason:\n"
    s << Formatter.error(reason)
    s << "\n" unless reason.end_with?("\n")
  end

  s.freeze
end
