def self.can_load?(ref)
  return false unless ref.respond_to?(:to_str)

  content = ref.to_str

  # Cache compiled regex
  @regex ||= begin
    token  = /(?:"[^"]*"|'[^']*')/
    curly  = /\(\s*#{token.source}\s*\)\s*\{.*\}/
    do_end = /\s+#{token.source}\s+do(?:\s*;\s*|\s+).*end/
    /\A\s*cask(?:#{curly.source}|#{do_end.source})\s*\Z/m
  end

  content.match?(@regex)
end
