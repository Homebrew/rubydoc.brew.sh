def env_method_name(env, hash)
  method_name = env.to_s
                   .sub(/^HOMEBREW_/, "")
                   .downcase
  method_name = "#{method_name}?" if hash[:boolean]
  method_name
end

def make_jobs
  jobs = ENV["HOMEBREW_MAKE_JOBS"].to_i
  return jobs.to_s if jobs.positive?

  ENVS.fetch(:HOMEBREW_MAKE_JOBS)
      .fetch(:default)
      .call
      .to_s
end
