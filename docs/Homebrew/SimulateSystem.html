def with(os: T.unsafe(nil), arch: T.unsafe(nil), &_block)
  raise ArgumentError, "At least one of `os` or `arch` must be specified." if !os && !arch

  old_os = self.os
  old_arch = self.arch

  begin
    self.os = os if os
    self.arch = arch if arch

    yield
  ensure
    @os = old_os
    @arch = old_arch
  end
end
