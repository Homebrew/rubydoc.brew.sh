def run
  ruby_sys_args = []
  ruby_sys_args << "-r#{args.r}" if args.r
  ruby_sys_args << "-e #{args.e}" if args.e
  ruby_sys_args += args.named

  exec(*HOMEBREW_RUBY_EXEC_ARGS,
       "-I", $LOAD_PATH.join(File::PATH_SEPARATOR),
       "-rglobal", "-rdev-cmd/irb",
       *ruby_sys_args)
end
