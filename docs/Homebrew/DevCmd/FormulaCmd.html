def run
  formula_paths = args.named.to_paths(only: :formula).select(&:exist?)
  if formula_paths.blank? && args.named
                                 .to_paths(only: :cask)
                                 .any?(&:exist?)
    odie "Found casks but did not find formulae!"
  end
  formula_paths.each { puts _1 }
end
