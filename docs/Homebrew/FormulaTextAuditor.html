def line_number(regex, skip = 0)
  index = @lines.drop(skip).index { |line| line =~ regex }
  index ? index + 1 : nil
end

def reverse_line_number(regex)
  index = @lines.reverse.index { |line| line =~ regex }
  index ? @lines.count - index : nil
end
