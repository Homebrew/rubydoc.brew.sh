def plist
  @plist ||= begin
    output = stdout

    if /\A(?<garbage>.*?)<\?\s*xml/m =~ output
      output = output.sub(/\A#{Regexp.escape(garbage)}/m, "")
      warn_plist_garbage(garbage)
    end

    if %r{<\s*/\s*plist\s*>(?<garbage>.*?)\Z}m =~ output
      output = output.sub(/#{Regexp.escape(garbage)}\Z/, "")
      warn_plist_garbage(garbage)
    end

    Plist.parse_xml(output)
  end
end
