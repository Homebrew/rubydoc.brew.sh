def self.dump(env, out = $stdout)
  keys = self.keys(env)
  keys -= %w[CC CXX OBJC OBJCXX] if env["CC"] == env["HOMEBREW_CC"]

  keys.each do |key|
    value = env.fetch(key)

    string = +"#{key}: #{value}"
    case key
    when "CC", "CXX", "LD"
      string << " => #{Pathname.new(value).realpath}" if value.present? && File.symlink?(value)
    end
    string.freeze
    out.puts string
  end
end
