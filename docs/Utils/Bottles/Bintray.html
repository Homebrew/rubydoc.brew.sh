def self.package(formula_name)
  package_name = formula_name.to_s.dup
  package_name.tr!("+", "x")
  package_name.sub!(/(.)@(\d)/, "\\1:\\2") # Handle foo@1.2 style formulae.
  package_name
end

def self.repository(tap = nil)
  if tap.nil? || tap.core_tap?
    "bottles"
  else
    "bottles-#{tap.repo}"
  end
end
