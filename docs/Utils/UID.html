def self.drop_euid(&_block)
  return yield if Process.euid == Process.uid

  original_euid = Process.euid
  begin
    Process::Sys.seteuid(Process.uid)
    yield
  ensure
    Process::Sys.seteuid(original_euid)
  end
end
