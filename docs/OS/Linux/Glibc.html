def system_version
  @system_version ||= begin
    version = Utils.popen_read("/usr/bin/ldd", "--version")[/ (\d+\.\d+)/, 1]
    if version
      Version.new version
    else
      Version::NULL
    end
  end
end

def version
  @version ||= begin
    version = Utils.popen_read(HOMEBREW_PREFIX/"opt/glibc/bin/ldd", "--version")[/ (\d+\.\d+)/, 1]
    if version
      Version.new version
    else
      system_version
    end
  end
end
