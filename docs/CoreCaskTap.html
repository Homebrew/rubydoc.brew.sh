def tap_migrations
  @tap_migrations ||= if Homebrew::EnvConfig.no_install_from_api?
    super
  else
    migrations, = Homebrew::API.fetch_json_api_file "cask_tap_migrations.jws.json",
                                                    stale_seconds: TAP_MIGRATIONS_STALE_SECONDS
    migrations
  end
