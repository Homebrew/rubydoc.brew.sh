def attr_rw(*attrs)
  attrs.each do |attr|
    module_eval <<-EOS, __FILE__, __LINE__+1
      def #{attr}(val=nil)                         # def prefix(val=nil)
        if val.nil?                                #   if val.nil?

          else                                     #      else

          end                                      #      end
        end                                        #    end

      end                                          # end
    EOS
  end
