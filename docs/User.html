def self.current
  return @current if defined?(@current)

  pwuid = Etc.getpwuid(Process.euid)
  return if pwuid.nil?

  @current = new(pwuid.name)
end

def gui?
  out, _, status = system_command "who"
  return false unless status.success?

  out.lines
     .map(&:split)
     .any? { |user, type,| user == self && type == "console" }
end
